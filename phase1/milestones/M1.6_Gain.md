# M1.6 增益函数 — 体积增益 + 目标偏置

## 目标

> 给一个候选 viewpoint v，算它的"探索价值" gain(v)。这是 M1.7 选最佳节点的核心打分。

```text
gain(v) = w_vol · volumetric_gain(v, voxel_map)
        + w_target · target_bias(v, P_weld)
```

- **volumetric_gain**：站在 v 看出去能扫到多少未观测体素（GBPlanner2 同款）。
- **target_bias**：相机方向是否朝 P_weld 中点（让 Phase 1 偏向目标，加速进入 Phase 2）。

## 在管线中的位置

```text
M1.7 单 cycle:
  for each candidate viewpoint v:
    g = gain(v, voxel_map, P_weld)        ← 这里
    score = g - λ · trajectory_cost      (距离惩罚)
  best = argmax score
```

## 关键设计决定

| 决定 | 理由 |
|------|------|
| 目标偏置用 cosine | 简单平滑；范围 [-1, 1] |
| 视锥用稀疏射线（不是全像素） | 算 gain 是估计值，不需要 720p；用 8×8 = 64 根射线足矣 |
| 用已有 voxel_map.raycast | 不重复实现 |
| 入射角不重复检查 | M1.3 已经做了硬约束；这里只算"价值" |

## 实施步骤

### Step 1: target_bias

```python
def target_bias(cam_pos, cam_dir, p_target):
    direction_to_target = p_target - cam_pos
    direction_to_target /= np.linalg.norm(direction_to_target)
    cam_dir_n = cam_dir / np.linalg.norm(cam_dir)
    return float(np.dot(cam_dir_n, direction_to_target))
```

### Step 2: volumetric_gain

```python
def volumetric_gain(cam_pos, cam_dir, cam_up, voxel_map, cfg):
    # 在视锥里撒 n_h × n_v 根射线
    H, V = cfg.n_rays_h, cfg.n_rays_v
    half_fov_h = np.radians(cfg.fov_h_deg / 2)
    half_fov_v = np.radians(cfg.fov_v_deg / 2)
    
    # 角度网格
    theta_h = np.linspace(-half_fov_h, half_fov_h, H)   # 水平偏角
    theta_v = np.linspace(-half_fov_v, half_fov_v, V)   # 垂直偏角
    th_h, th_v = np.meshgrid(theta_h, theta_v, indexing='ij')
    
    # 相机系射线方向 (z 朝前)
    dx = np.tan(th_h)
    dy = np.tan(th_v)
    dz = np.ones_like(dx)
    dirs_cam = np.stack([dx, dy, dz], axis=-1)
    dirs_cam /= np.linalg.norm(dirs_cam, axis=-1, keepdims=True)
    
    # camera frame → world: 用 cam_dir, cam_up 构造 R
    z_c = cam_dir / np.linalg.norm(cam_dir)
    x_c = np.cross(cam_up, z_c); x_c /= np.linalg.norm(x_c)
    y_c = np.cross(z_c, x_c)
    R = np.stack([x_c, y_c, z_c], axis=1)  # 列是 c 系基向量
    dirs_world = (R @ dirs_cam.reshape(-1, 3).T).T   # (H*V, 3)
    
    origins = np.broadcast_to(cam_pos, dirs_world.shape).copy()
    
    origins_t = torch.from_numpy(origins).to(voxel_map.device).float()
    dirs_t = torch.from_numpy(dirs_world).to(voxel_map.device).float()
    
    out = voxel_map.raycast(origins_t, dirs_t, max_range=cfg.max_range)
    total_unknown = int(out['unknown_count'].sum().item())
    return float(total_unknown)
```

### Step 3: 批量化

```python
def gain_batch(cam_poses, voxel_map, p_target, cfg):
    """暂时朴素循环. 性能不够再 vectorize."""
    out = torch.zeros(len(cam_poses), device=voxel_map.device)
    for i, T in enumerate(cam_poses.cpu().numpy()):
        cam_pos = T[:3, 3]
        cam_dir = T[:3, 2]    # z_camera
        cam_up  = -T[:3, 1]   # 相机 y 朝下, up 是 -y
        out[i] = gain(cam_pos, cam_dir, cam_up, voxel_map, p_target, cfg)
    return out
```

性能瓶颈：`gain_batch` 每个 candidate 都做 64 个 raycast（M1.2 内部已经向量化）。整个 batch 的 raycast 可以一次并发——把 N×64 根射线一起喂给 voxel_map.raycast。可以在第二轮优化做。

```python
# tests/test_gain.py

def test_target_bias_aligned():
    """相机正对目标, bias = +1."""
    cam_pos = np.array([0, 0, 0])
    p_target = np.array([0, 0, 1])
    cam_dir = np.array([0, 0, 1])
    assert abs(target_bias(cam_pos, cam_dir, p_target) - 1.0) < 1e-6


def test_target_bias_opposite():
    """相机背对目标, bias = -1."""
    cam_pos = np.array([0, 0, 0])
    p_target = np.array([0, 0, 1])
    cam_dir = np.array([0, 0, -1])
    assert abs(target_bias(cam_pos, cam_dir, p_target) - (-1.0)) < 1e-6


def test_target_bias_orthogonal():
    """相机和目标垂直, bias = 0."""
    cam_pos = np.array([0, 0, 0])
    p_target = np.array([0, 0, 1])
    cam_dir = np.array([1, 0, 0])
    assert abs(target_bias(cam_pos, cam_dir, p_target)) < 1e-6


def test_volumetric_gain_empty_map():
    """全 unknown 地图, 任何 viewpoint 的 gain 都接近 max_count."""
    vm = VoxelMap(bounds=np.array([[-1,-1,-1],[1,1,1]]), resolution=0.05)
    # 全 unknown
    cam_pos = np.array([0, 0, 0])
    cam_dir = np.array([0, 0, 1])
    cam_up = np.array([0, -1, 0])  # OpenCV: y 朝下
    cfg = GainConfig(n_rays_h=8, n_rays_v=8, max_range=1.0)
    g = volumetric_gain(cam_pos, cam_dir, cam_up, vm, cfg)
    # 64 射线, 每根穿 1m / 0.05m = 20 体素 → ~ 1280
    assert g > 1000


def test_volumetric_gain_known_free():
    """全 free 地图, gain = 0."""
    vm = VoxelMap(bounds=np.array([[-1,-1,-1],[1,1,1]]), resolution=0.05)
    vm._state[:] = 1
    g = volumetric_gain(np.zeros(3), np.array([0,0,1]), np.array([0,-1,0]), vm,
                       GainConfig(max_range=1.0))
    assert g == 0


def test_volumetric_gain_blocked():
    """前面有墙挡, gain 远小于全 unknown."""
    vm = VoxelMap(bounds=np.array([[-1,-1,-1],[1,1,1]]), resolution=0.05)
    # 前面 0.2m 处一面 occupied 墙
    vm._state[..., int((0.2 + 1) / 0.05)] = 2
    g_blocked = volumetric_gain(np.zeros(3), np.array([0,0,1]), np.array([0,-1,0]), vm,
                                GainConfig(max_range=1.0))
    
    vm2 = VoxelMap(bounds=np.array([[-1,-1,-1],[1,1,1]]), resolution=0.05)
    g_open = volumetric_gain(np.zeros(3), np.array([0,0,1]), np.array([0,-1,0]), vm2,
                             GainConfig(max_range=1.0))
    
    assert g_blocked < g_open / 2


def test_gain_combines():
    """gain = w1*vol + w2*target."""
    vm = VoxelMap(bounds=np.array([[-1,-1,-1],[1,1,1]]), resolution=0.05)
    cfg = GainConfig(w_vol=1.0, w_target=0.0, n_rays_h=4, n_rays_v=4, max_range=1.0)
    g_vol_only = gain(np.zeros(3), np.array([0,0,1]), np.array([0,-1,0]),
                     vm, np.array([0,0,1]), cfg)
    
    cfg.w_target = 100.0
    g_with_target = gain(np.zeros(3), np.array([0,0,1]), np.array([0,-1,0]),
                        vm, np.array([0,0,1]), cfg)
    # target = +1, 加 100, 应该比单纯 vol 大约 +100
    assert abs(g_with_target - g_vol_only - 100) < 1.0
```

```python
def test_gain_with_real_beam_seam():
    """加载 BEAM 工件, 喂进 voxel map, 算几个不同 viewpoint 的 gain.
    朝向 P_weld 的 viewpoint 应比朝向背面的 gain 大."""
    vm = VoxelMap(bounds=BEAM_WORKSPACE_BOUNDS)
    rsim = RaycastSim(mesh_paths=[BEAM_OBJ])
    # 喂几帧, 让地图有点东西
    for pose in INITIAL_VIEWS:
        depth = rsim.render(pose)
        vm.integrate(depth, INTRIN, pose)
    
    p_weld = np.array([24.030, 0.157, 11.432])
    
    # 朝 P_weld 看
    cam_facing = np.array([24.4, 0.5, 11.0])
    dir_facing = (p_weld - cam_facing) / np.linalg.norm(p_weld - cam_facing)
    
    # 背对 P_weld
    cam_back = np.array([24.4, 0.5, 11.0])
    dir_back = -dir_facing
    
    g_facing = gain(cam_facing, dir_facing, np.array([0,0,1]), vm, p_weld)
    g_back = gain(cam_back, dir_back, np.array([0,0,1]), vm, p_weld)
    
    print(f"facing: {g_facing}, back: {g_back}")
    assert g_facing > g_back
```
