# M1.2 PyTorch 体素栅格 — 在线建图

## 目标

> 维护一个 GPU 上的三态体素栅格 (free / occupied / unknown)，用 RGB-D 帧不断更新。提供 raycast 查询能力。

## 在管线中的位置

```text
RaycastSim.render(pose) ─→ depth (H, W)
                            │
                            ▼
                 VoxelMap.integrate(depth, K, pose)
                            │
              ┌─────────────┼──────────────┐
              ▼             ▼              ▼
        observation_  volumetric_    arm_kin
        check         gain           collision
        (M1.3)        (M1.6)         (M1.4)
```

## 关键设计决定

| 决定 | 选项 | 选 | 理由 |
|------|------|------|------|
| 数据结构 | dense 3D tensor vs hash | dense | 工作空间小（1.5 m³ × 0.02 res = 75³ ≈ 42 万体素），dense 简单 |
| 设备 | CPU vs GPU | GPU | raycast/integrate 都在 GPU 跑，避免数据搬运 |
| 状态编码 | 概率 vs 三态 | uint8 三态 (0/1/2) | Phase 1 不需要概率融合；将来可换 |
| 工作空间 | 全场景 vs 局部 | 全场景固定 | 工件 + 周围 1 m，固定 bounds 简单 |
| Raycast 算法 | 步进 vs Amanatides-Woo | 步进 (DDA-like) | 实现 ~50 行；精确 voxel traversal 留作优化 |
| 自由空间标记 | 整条射线 free vs 仅终点 | 整条射线 free | OctoMap 标准做法 |

## 接口规范

```python
# phase1/mapping.py

import torch
import numpy as np
from typing import Literal


VoxelState = Literal[0, 1, 2]  # 0=unknown, 1=free, 2=occupied


class VoxelMap:
    """三态稠密体素栅格. 全部数据在 GPU."""

    def __init__(
        self,
        bounds: np.ndarray,          # (2, 3) [[xmin,ymin,zmin],[xmax,ymax,zmax]]
        resolution: float = 0.02,    # 米
        device: str = "cuda:0",
        max_range: float = 2.0,      # raycast 最远多少米
    ):
        """初始化 unknown 体素栅格."""
        ...

    @property
    def shape(self) -> tuple[int, int, int]:
        """(Nx, Ny, Nz)."""
        ...

    @property
    def state(self) -> torch.Tensor:
        """(Nx, Ny, Nz) uint8."""
        ...

    def world_to_index(self, pts: torch.Tensor) -> torch.Tensor:
        """(N, 3) world 坐标 → (N, 3) int64 体素索引. 越界返回 -1."""
        ...

    def index_to_world(self, idx: torch.Tensor) -> torch.Tensor:
        """(N, 3) 索引 → (N, 3) 体素中心 world 坐标."""
        ...

    def query(self, pts: torch.Tensor) -> torch.Tensor:
        """(N, 3) world 坐标 → (N,) uint8 状态. 越界返回 0 (unknown)."""
        ...

    def integrate(
        self,
        depth: torch.Tensor,         # (H, W) 深度图 (米), 0=没 hit
        intrinsics: "CameraIntrinsics",
        camera_pose: np.ndarray,     # (4, 4) world→camera
    ) -> None:
        """喂一帧深度, 更新内部 state.
        每个像素:
          - 沿射线从 origin 到 hit 之间的体素标 free
          - hit 体素本身标 occupied
          - hit 后面的体素不动 (unknown 还是 unknown)
          - 深度 = 0 (没 hit) 时, 沿 max_range 全标 free
        """
        ...

    def raycast(
        self,
        origins: torch.Tensor,       # (N, 3) world
        directions: torch.Tensor,    # (N, 3) 单位向量 world
        max_range: float | None = None,
    ) -> dict:
        """N 根射线一起做 raycast. 返回:
        {
          'hit': (N,) bool,
          'distance': (N,) float, 没 hit 是 max_range
          'unknown_count': (N,) int, 沿射线穿过的 unknown 体素数
          'first_unknown': (N,) float, 第一个 unknown 体素的距离 (没有就 max_range)
        }
        """
        ...

    def num_voxels_by_state(self) -> dict[str, int]:
        """统计三态各多少, 方便 debug. {'unknown': N1, 'free': N2, 'occupied': N3}."""
        ...
```

## 实施步骤

### Step 1: 初始化

```python
def __init__(self, bounds, resolution, device, max_range):
    self.bounds = torch.tensor(bounds, device=device, dtype=torch.float32)  # (2, 3)
    self.res = float(resolution)
    self.device = device
    self.max_range = max_range
    nx = int(np.ceil((bounds[1, 0] - bounds[0, 0]) / resolution))
    ny = int(np.ceil((bounds[1, 1] - bounds[0, 1]) / resolution))
    nz = int(np.ceil((bounds[1, 2] - bounds[0, 2]) / resolution))
    self._shape = (nx, ny, nz)
    self._state = torch.zeros((nx, ny, nz), dtype=torch.uint8, device=device)
```

### Step 2: 坐标转换

```python
def world_to_index(self, pts):
    rel = (pts - self.bounds[0]) / self.res
    idx = rel.long()
    mask = ((idx >= 0) & (idx < torch.tensor(self._shape, device=pts.device))).all(-1)
    idx[~mask] = -1
    return idx

def index_to_world(self, idx):
    return self.bounds[0] + (idx.float() + 0.5) * self.res
```

### Step 3: integrate（关键）

把深度图反投成 3D 点云（射线终点），然后沿射线步进标记。

```python
def integrate(self, depth, intrin, cam_pose):
    H, W = depth.shape
    # 1. 像素 → 相机系射线方向
    u = torch.arange(W, device=depth.device).float()
    v = torch.arange(H, device=depth.device).float()
    vv, uu = torch.meshgrid(v, u, indexing='ij')
    x = (uu - intrin.cx) / intrin.fx
    y = (vv - intrin.cy) / intrin.fy
    z = torch.ones_like(x)
    dirs_cam = torch.stack([x, y, z], -1)              # (H, W, 3)
    dirs_cam = dirs_cam / dirs_cam.norm(dim=-1, keepdim=True)
    
    # 2. 变换到 world
    R = torch.tensor(cam_pose[:3, :3], device=depth.device, dtype=torch.float32)
    t = torch.tensor(cam_pose[:3, 3], device=depth.device, dtype=torch.float32)
    dirs_world = (R @ dirs_cam.reshape(-1, 3).T).T     # (H*W, 3)
    origins = t.unsqueeze(0).expand_as(dirs_world)     # (H*W, 3)
    depths_flat = depth.flatten()                      # (H*W,)
    
    # 3. 沿每条射线步进
    n_steps = int(self.max_range / self.res) + 1
    step_dist = torch.linspace(self.res / 2, self.max_range, n_steps, device=depth.device)  # (S,)
    # 第 s 步上的点: origins[..., None, :] + dirs[..., None, :] * step_dist[None, :, None]
    # shape (H*W, S, 3)
    points = origins.unsqueeze(1) + dirs_world.unsqueeze(1) * step_dist.view(1, -1, 1)
    
    # 4. 区分 free / hit / 后面
    hit_dist = depths_flat.unsqueeze(1)                # (H*W, 1)
    no_hit = (hit_dist == 0)                           # (H*W, 1)
    # 没 hit 的射线: 整条标 free 到 max_range
    is_free = (step_dist.unsqueeze(0) < hit_dist) | no_hit
    is_occupied = (~no_hit) & (
        (step_dist.unsqueeze(0) >= hit_dist) &
        (step_dist.unsqueeze(0) < hit_dist + self.res)
    )
    # is_occupied 只取每条射线的一个体素 (最接近 hit 的)
    # is_free: 之前的全部
    
    # 5. world → idx, 写状态
    idx = self.world_to_index(points.reshape(-1, 3))   # (H*W*S, 3)
    valid = (idx >= 0).all(-1)
    flat_idx = idx[valid]
    flat_free = is_free.flatten()[valid]
    flat_occ  = is_occupied.flatten()[valid]
    
    # 写: 优先级 occupied > free (occupied 不被 free 覆盖)
    self._state[flat_idx[flat_free, 0], flat_idx[flat_free, 1], flat_idx[flat_free, 2]] = 1
    self._state[flat_idx[flat_occ,  0], flat_idx[flat_occ,  1], flat_idx[flat_occ,  2]] = 2
```

实现注意：

- 用 `torch.scatter_` 避免索引冲突
- occupied 的写入要在 free 之后，否则 hit 体素会被它前一步的 free 覆盖
- 内存：(H × W × S × 3) ≈ (720 × 1280 × 100 × 3) × 4B = 1.1 GB。**这太大了**，需要分块处理（按行块）

修订：每次 integrate 把像素分成若干 chunk（如每 chunk 50000 像素），循环处理。详细实现里会处理。

def test_empty_state():
    """全 unknown 时, query 任意点都是 0."""
    vm = VoxelMap(bounds=np.array([[0,0,0],[1,1,1]]), resolution=0.1)
    pts = torch.rand(100, 3, device='cuda')

def test_integrate_wall():
    """墙在 z=1, 相机在 z=0 朝 +z 看. 深度全部 = 1.
    预期: z<1 全 free, z=1 一层 occupied, z>1 全 unknown."""
    vm = VoxelMap(bounds=np.array([[-0.5,-0.5,0],[0.5,0.5,2]]), resolution=0.05)
    depth = torch.ones((100, 100), device='cuda')
    intrin = CameraIntrinsics.from_fov(90, 90, 100, 100)
    cam_pose = np.eye(4)  # 相机在原点朝 +z
    vm.integrate(depth, intrin, cam_pose)
    
    # 检查 z=0.5 处应该是 free
    pts_free = torch.tensor([[0, 0, 0.5]], device='cuda')
    assert vm.query(pts_free).item() == 1
    # 检查 z=1.0 处应该是 occupied
    pts_occ = torch.tensor([[0, 0, 1.0]], device='cuda')
