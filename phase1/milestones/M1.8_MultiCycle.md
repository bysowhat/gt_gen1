# M1.8 多 cycle + 卡住回退

## 目标

> 把 M1.7 的单 cycle 包成完整 loop：每 cycle 拍 depth → 建图 → 调 M1.7 → 执行 → 检测切换。再加上 GBPlanner2 风格的"卡住 → Global step"回退。

## 在管线中的位置

```text
run_phase1():    # M1.10
  state = init()
  while not done:
    if local_stuck(state):
      do_global_step(state)         ← M1.8 新增
    else:
      do_local_step(state)          ← M1.7 单 cycle
    state.cycle_idx += 1
```

## 与 M1.7 的差别

| | M1.7 单 cycle | M1.8 多 cycle |
|---|---------------|---------------|
| 范围 | 1 步决策 | 完整任务 (探到为止) |
| 状态 | 无状态 | stuck_count, last_node_id, etc. |
| Dead-end 处理 | 无 | 有 (Global step) |
| 终止条件 | 单步执行完 | phase_switch / 全图无 frontier / max_cycles |

## Global Step 详细流程

```text
触发: 连续 K_STUCK (=3) 次 cycle 的 best_score < g_low

执行:
  Step 1  在全局图里所有节点上, 重算 volumetric_gain (因为地图变了)
  Step 2  挑出 gain > g_threshold 的节点 = "frontier nodes"
  Step 3  用 Dijkstra 算从当前位置到每个 frontier 的最短路径 cost
  Step 4  挑 score = gain · exp(-λ·cost) 最高的 frontier
  Step 5  执行整条 Dijkstra 路径 (不是单边, 因为这条路是验证过的 free)
  Step 6  到达后回到 Local 模式
  
若全图都没 frontier: 整个任务失败, 报告
```

## 关键设计决定

| 决定 | 理由 |
|------|------|
| 重算 gain 时只看 graph 节点 | 比重新撒采样省事；只是在已知好位姿上重评 |
| Global 时执行整条路径 | 已知 free，不需要 step-by-step；安全 |
| 卡住计数器要在 Local 失败 / Global 失败时都生效 | 否则可能一直来回切 |
| 加最大 cycle 限制 | 防止真的探索失败时无限循环 |

## 实施步骤

### Step 1: 主 loop

```python
def run_phase1(theta_init, seam_points, seam_tangents, p_weld_center,
               arm_kin, depth_source, intrinsics, cfg, visualizer):
    import time
    t_start = time.time()
    
    # 初始化 state
    state = Phase1State(
        theta_now=theta_init.copy(),
        voxel_map=VoxelMap(bounds=cfg_to_bounds(cfg, p_weld_center),
                           resolution=cfg.voxel_res, max_range=cfg.voxel_max_range),
        graph=ExplorationGraph(),
        arm_kin=arm_kin,
        seam_points=seam_points,
        seam_tangents=seam_tangents,
        p_weld_center=p_weld_center,
        cfg=cfg,
        intrinsics=intrinsics,
    )
    
    # 加初始节点
    fk = arm_kin.fk(torch.tensor(theta_init).unsqueeze(0).cuda())
    init_node = GraphNode(
        node_id=str(uuid.uuid4()),
        theta=theta_init,
        ee_pos_world=fk['ee_pos_world'][0].cpu().numpy(),
        ee_dir_world=fk['ee_quat_world'][0].cpu().numpy(),  # 简化, 实际从 quat 反推
        cycle_added=0,
    )
    state.graph.add_node(init_node)
    state.theta_now = theta_init
    
    while True:
        # 安全终止
        if state.cycle_idx >= cfg.max_cycles:
            return Phase1Result(
                success=False, fail_reason="max_cycles_reached", ...)
        
        # 拍 depth + 建图
        cur_ee_pose = compute_camera_pose(state.theta_now, arm_kin)
        depth = depth_source.render(cur_ee_pose)
        state.voxel_map.integrate(depth, intrinsics, cur_ee_pose)
        
        # 检查切换
        cur_vp = camera_pose_to_viewpoint(cur_ee_pose)
        observed_now = detect_phase_switch(
            state.voxel_map, seam_points, seam_tangents,
            cur_vp, cfg, intrinsics,
        )
        state.p_observed.update(observed_now)
        
        if state.p_observed:
            return Phase1Result(
                success=True,
                phase_switch_at=state.cycle_idx,
                p_observed_first=min(state.p_observed),
                total_time_s=time.time() - t_start,
                ...
            )
        
        # 决策: local 还是 global?
        if is_local_stuck(state):
            ok = do_global_step(state, depth_source, visualizer)
            if not ok:
                return Phase1Result(success=False, fail_reason="no_frontier", ...)
            state.last_global_cycle = state.cycle_idx
            state.stuck_count = 0
        else:
            cycle_result = run_one_cycle(
                state.theta_now, state.voxel_map, state.graph,
                state.arm_kin, state.seam_points, state.seam_tangents,
                state.p_weld_center, state.cycle_idx, state.cfg, state.intrinsics,
            )
            state.cycle_history.append(cycle_result)
            
            if not cycle_result.success or cycle_result.best_score < cfg.g_low:
                state.stuck_count += 1
            else:
                state.stuck_count = 0
                state.theta_now = cycle_result.next_theta
        
        if visualizer is not None:
            visualizer.update(state)
        
        state.cycle_idx += 1
```

### Step 2: do_global_step

```python
def do_global_step(state, depth_source, visualizer):
    cfg = state.cfg
    
    # 重算所有节点的 gain
    reevaluate_node_gains(state.graph, state.voxel_map, state.p_weld_center, cfg)
    
    # 找当前节点 ID
    cur_node = state.graph.nearest_node(state.theta_now)
    
    # 找 frontier
    f_star = pick_frontier_node(state.graph, state.theta_now, cfg)
    if f_star is None:
        return False
    
    # Dijkstra
    path = state.graph.dijkstra(cur_node.node_id, f_star.node_id)
    if path is None:
        return False
    
    # 沿 path 一步步移动 + 拍 depth + 整合
    for next_node in path[1:]:
        # 简化: 假设 path 上每条边在地图变化前都是 free 的
        # 严格起见应该 re-check
        cur_ee_pose = compute_camera_pose(next_node.theta, state.arm_kin)
        depth = depth_source.render(cur_ee_pose)
        state.voxel_map.integrate(depth, state.intrinsics, cur_ee_pose)
        state.theta_now = next_node.theta
        
        if visualizer is not None:
            visualizer.update(state)
    
    return True


def pick_frontier_node(graph, theta_now, cfg):
    fronts = graph.frontier_nodes(gain_threshold=cfg.g_low * 2)
    if not fronts:
        return None
    cur_node = graph.nearest_node(theta_now)
    if cur_node is None:
        return None
    
    # 算到每个 frontier 的距离 + 综合 score
    cost_map = graph.dijkstra_to_many(cur_node.node_id,
                                       [f.node_id for f in fronts])
    best, best_score = None, -1e9
    for f in fronts:
        if f.node_id not in cost_map:
            continue
        path = cost_map[f.node_id]
        cost = sum(np.linalg.norm(path[i].theta - path[i+1].theta)
                   for i in range(len(path)-1))
        score = f.last_gain * np.exp(-cfg.lambda_cost * cost)
        if score > best_score:
            best_score = score
            best = f
    return best
```

### Step 3: reevaluate_node_gains

```python
def reevaluate_node_gains(graph, voxel_map, p_weld_center, cfg):
    """对图里每个节点, 重算 volumetric_gain."""
    gain_cfg = GainConfig(
        w_vol=cfg.w_vol, w_target=cfg.w_target,
        n_rays_h=8, n_rays_v=8, max_range=cfg.voxel_max_range,
    )
    for nid in list(graph._g.nodes()):
        node = graph.get_node(nid)
        new_g = compute_gain(
            node.ee_pos_world,
            node.ee_dir_world / np.linalg.norm(node.ee_dir_world),
            np.array([0, 0, 1]),  # 简化, 实际存 up
            voxel_map, p_weld_center, gain_cfg,
        )
        node.last_gain = new_g
```

def test_pick_frontier_picks_high_gain_close():
    """两个 frontier, 一个近 gain 中, 一个远 gain 高. 应选综合高的."""
    g = ExplorationGraph()
    # 建一个 5 节点的链
    nodes = []
    for i in range(5):
        n = GraphNode(f"n{i}", np.full(6, i*0.1), np.zeros(3),
                      np.array([0,0,1]), 0)
        n.last_gain = 0.0
        g.add_node(n)
        nodes.append(n)
        if i > 0:
            g.add_edge(f"n{i-1}", f"n{i}", weight=0.1*np.sqrt(6))
    
    # n1 gain=20 (近), n4 gain=100 (远)
    nodes[1].last_gain = 20
    nodes[4].last_gain = 100
    
    cfg = Phase1Config(lambda_cost=0.5, g_low=2)
    f = pick_frontier_node(g, theta_now=np.zeros(6), cfg=cfg)
    # cost 短的是 n1: 0.1√6 ≈ 0.245, score = 20·exp(-0.5·0.245) = 17.7
    # cost 长的是 n4: 0.4√6 ≈ 0.98,   score = 100·exp(-0.5·0.98) = 61.3
    assert f.node_id == "n4"
