# M1.3 合格观测检查器 — 4 条硬约束

## 目标

> 给定一个 viewpoint v 和一个焊缝采样点 P_i，判断 v 是否"合格观测" P_i。

## 在管线中的位置

```text
单 cycle (M1.7):
  for each P_i ∉ S:
    for each candidate viewpoint v:
      if is_valid_observation(v, P_i, voxel_map):  ←─ M1.3
        ...
        
切换条件 (M1.7+):
  if any P_i passes is_valid_observation():
    Phase 1 → Phase 2
```

## 4 条硬约束（出自焊缝场景设计文档）

```text
viewpoint v = (camera_pos, camera_dir, camera_up) 合格观测 P_i 需要全部满足:

  ① 距离:    d_min ≤ ‖camera_pos - P_i‖ ≤ d_max
            (默认 d_min=0.3m, d_max=0.6m)
  
  ② 视锥内:  P_i 在相机的视场角内
            投影到图像 (u,v), 0 ≤ u < W 且 0 ≤ v < H, 且 z>0
  
  ③ 视线无障: raycast(camera_pos, P_i 方向) 直到 P_i 距离都不被 occupied 体素挡
            unknown 体素也不允许 (否则赌博)
  
  ④ 入射角:  视线方向与焊缝局部切线 t_i 的夹角 ∈ [30°, 90°]
            避免相机看焊缝的"侧面 (与切线平行)"
            数学: angle = arccos(|cam_dir · tangent|)
```

## 关键设计决定

| 决定 | 理由 |
|------|------|
| 4 条返回独立结果 | 调试时知道哪条挂了 |
| `unknown` 视线视作障碍 | M1.7 之后，Phase 2 才允许穿 unknown 探出去 |
| 入射角用绝对值 | 切线方向有正反两个，取小角 |
| 距离用相机位置（不是末端关节）| 末端有相机偏置，这里用相机本身的位置 |

## 各条实现细节

### 条件 ① 距离

```python
def check_distance(viewpoint, p_seam, d_min, d_max):
    d = float(np.linalg.norm(viewpoint.pos - p_seam))
    return d_min <= d <= d_max, d
```

### 条件 ② 视锥内

把 P_i 投影到相机系，看像素是否在 (0, W) × (0, H) 内：

```python
def check_in_frustum(viewpoint, p_seam, intrin):
    # world → camera frame
    rel = p_seam - viewpoint.pos                    # (3,)
    # 构造 R_world_cam: cam_dir 是 z_c
    z_c = viewpoint.dir / np.linalg.norm(viewpoint.dir)
    # 选一个临时 up; 严格起见用 viewpoint.up. 暂时用世界 z 反, 投影后 normalize.
    up = viewpoint.up if viewpoint.up is not None else np.array([0, 0, 1], dtype=float)
    x_c = np.cross(up, z_c); x_c /= np.linalg.norm(x_c)
    y_c = np.cross(z_c, x_c)
    R_wc = np.stack([x_c, y_c, z_c], axis=1)        # world→cam: rel_cam = R_wc.T @ rel
    rel_cam = R_wc.T @ rel
    if rel_cam[2] <= 0:
        return False                                # 在相机后面
    u = intrin.fx * rel_cam[0] / rel_cam[2] + intrin.cx
    v = intrin.fy * rel_cam[1] / rel_cam[2] + intrin.cy
    return 0 <= u < intrin.width and 0 <= v < intrin.height
```

### 条件 ③ 视线无障

```python
def check_line_of_sight(viewpoint, p_seam, voxel_map, unknown_as_block):
    direction = p_seam - viewpoint.pos
    distance = np.linalg.norm(direction)
    direction = direction / distance
    
    origins = torch.from_numpy(viewpoint.pos.reshape(1, 3)).to(voxel_map.device).float()
    dirs    = torch.from_numpy(direction.reshape(1, 3)).to(voxel_map.device).float()
    out = voxel_map.raycast(origins, dirs, max_range=distance - voxel_map.res * 0.5)
    # 如果在 distance 之前就 hit 了 occupied → 挡住
    if out['hit'][0] and out['distance'][0].item() < distance - voxel_map.res:
        return False
    if unknown_as_block and out['unknown_count'][0].item() > 0:
        return False
    return True
```

### 条件 ④ 入射角

```python
def check_incidence(viewpoint, p_seam, tangent, ang_min, ang_max):
    direction = p_seam - viewpoint.pos
    direction = direction / np.linalg.norm(direction)
    tangent_n = tangent / np.linalg.norm(tangent)
    cos_a = abs(float(np.dot(direction, tangent_n)))
    angle_deg = np.degrees(np.arccos(np.clip(cos_a, -1, 1)))
    # angle 与 tangent 夹角. 但我们要相机视线和焊缝表面的关系.
    # 焊缝是一条曲线, "看焊缝的侧面" = 视线和切线平行 (angle ≈ 0).
    # 想要 angle ∈ [ang_min, ang_max].
    return ang_min <= angle_deg <= ang_max, angle_deg
```

### 主函数

```python
def is_valid_observation(viewpoint, p_seam, tangent, voxel_map, intrinsics,
                         d_min=0.3, d_max=0.6,
                         incidence_min_deg=30.0, incidence_max_deg=90.0,
                         los_unknown_as_block=True):
    dist_ok, dist = check_distance(viewpoint, p_seam, d_min, d_max)
    if not dist_ok:
        return ObservationResult(False, dist_ok, False, False, False,
                                 dist, 0.0, 'distance')
    
    frustum_ok = check_in_frustum(viewpoint, p_seam, intrinsics)
    if not frustum_ok:
        return ObservationResult(False, dist_ok, False, False, False,
                                 dist, 0.0, 'frustum')
    
    los_ok = check_line_of_sight(viewpoint, p_seam, voxel_map, los_unknown_as_block)
    if not los_ok:
        return ObservationResult(False, dist_ok, frustum_ok, False, False,
                                 dist, 0.0, 'line_of_sight')
    
    inc_ok, inc_angle = check_incidence(viewpoint, p_seam, tangent,
                                         incidence_min_deg, incidence_max_deg)
    if not inc_ok:
        return ObservationResult(False, dist_ok, frustum_ok, los_ok, False,
                                 dist, inc_angle, 'incidence')
    
    return ObservationResult(True, True, True, True, True, dist, inc_angle, None)
```

def make_test_setup():
    """构造一个简单场景: 焊缝点在 (0,0,0), 切线 (1,0,0), 相机能看到, 视线无障."""
    p_seam = np.array([0, 0, 0])
    tangent = np.array([1, 0, 0])
    
    # 一个 1m³ 的 free 工作空间, 没有障碍
    vm = VoxelMap(bounds=np.array([[-1,-1,-1],[1,1,1]]), resolution=0.05)
    # 整个空间标 free (没障碍)
    vm._state[:] = 1
    
    intrin = CameraIntrinsics.from_fov(60, 45, 480, 640)
    return p_seam, tangent, vm, intrin


def test_valid_observation():
    """相机在 (0, 0, 0.45) 朝 -z 看 P_i, 4 条都过."""
    p_seam, tangent, vm, intrin = make_test_setup()
    vp = Viewpoint(
        pos=np.array([0, 0.4, 0.0]),
        dir=np.array([0, -1, 0]),
        up=np.array([0, 0, 1]),
    )
    result = is_valid_observation(vp, p_seam, tangent, vm, intrin)
    assert result.valid, f"failed at: {result.fail_reason}"


def test_too_close():
    """距离 < d_min."""
    p_seam, tangent, vm, intrin = make_test_setup()
    vp = Viewpoint(pos=np.array([0, 0.1, 0.0]),  # 0.1m < d_min=0.3
                   dir=np.array([0, -1, 0]),
                   up=np.array([0, 0, 1]))
    result = is_valid_observation(vp, p_seam, tangent, vm, intrin)
    assert not result.valid
    assert result.fail_reason == 'distance'


def test_too_far():
    """距离 > d_max."""
    ...


def test_behind_camera():
    """P_i 在相机背后, frustum 失败."""
    p_seam, tangent, vm, intrin = make_test_setup()
    vp = Viewpoint(pos=np.array([0, 0.4, 0]),
                   dir=np.array([0, +1, 0]),  # 朝 +y, P_i 在 -y 方向, 在背后
                   up=np.array([0, 0, 1]))
    result = is_valid_observation(vp, p_seam, tangent, vm, intrin)
    assert not result.valid
    assert result.fail_reason == 'frustum'


def test_blocked_by_occupied():
    """中间放个柱子挡住视线."""
    p_seam, tangent, vm, intrin = make_test_setup()
    # 把 (0, 0.2, 0) 附近一个体素标 occupied
    vm._state[20, 24, 20] = 2  # 假设这个 idx 对应 (0, 0.2, 0)
    vp = Viewpoint(pos=np.array([0, 0.4, 0]),
                   dir=np.array([0, -1, 0]),
                   up=np.array([0, 0, 1]))
    result = is_valid_observation(vp, p_seam, tangent, vm, intrin)
    assert not result.valid
    assert result.fail_reason == 'line_of_sight'


def test_incidence_too_parallel():
    """视线和切线接近平行 (相机沿焊缝方向看), incidence 失败."""
    p_seam, tangent, vm, intrin = make_test_setup()
    # tangent = (1, 0, 0), 视线也朝 +x
    vp = Viewpoint(pos=np.array([-0.4, 0, 0]),
                   dir=np.array([1, 0, 0]),
                   up=np.array([0, 0, 1]))
    result = is_valid_observation(vp, p_seam, tangent, vm, intrin)
    assert not result.valid
    assert result.fail_reason == 'incidence'
