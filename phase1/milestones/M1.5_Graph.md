# M1.5 全局图 + Dijkstra

## 目标

> 维护 GBPlanner2 风格的持久化全局图 G(V, E)：节点是已访问/已采样的关节构型，边是已碰撞验证的连接。提供 Dijkstra 最短路径查询。

## 在管线中的位置

```text
M1.7 单 cycle:
  graph.add_node(theta_now)          ← 当前位置
  candidates = sample(...)
  for c in candidates:
    if arm_kin.edge_free(theta_now, c.theta, vm):
      graph.add_edge(theta_now, c.theta, weight=...)
  
  best = argmax score
  path = graph.dijkstra(theta_now, best)
  执行 path[1] (第一条边)

M1.8 dead-end recovery:
  if stuck:
    f_star = pick_global_frontier(graph, voxel_map)
    path = graph.dijkstra(theta_now, f_star)
    sequentially execute path
```

## 关键设计决定

| 决定 | 选项 | 选 | 理由 |
|------|------|------|------|
| 图库 | networkx vs 自实现 | networkx | 已装；6 维节点 + Dijkstra 性能够 |
| 节点 ID | hash(tuple) vs uuid | uuid (str) | 浮点 hash 不稳；string 可以做 metadata 索引 |
| 节点附加属性 | dict vs dataclass | dataclass | 类型安全；后期 IDE 友好 |
| 距离度量 | cuRobo cost vs L2 | 从 cuRobo 拿 | 关节空间 L2 即可；权重靠 ArmKin |
| 持久化 | 内存 vs 磁盘 | 内存 | Phase 1 不需要 cross-session |
| 节点近邻查询 | scipy KDTree vs 暴力 | sklearn BallTree | 6 维 KDTree 退化, BallTree 好 |

```python
# phase1/graph.py

import uuid
from dataclasses import dataclass, field
from typing import Optional

import networkx as nx
import numpy as np
import torch
from sklearn.neighbors import BallTree


@dataclass
class GraphNode:
    """节点元数据."""
    node_id: str                   # uuid
    theta: np.ndarray              # (6,) 关节角
    ee_pos_world: np.ndarray       # (3,) 末端 world 系位置
    ee_dir_world: np.ndarray       # (3,) 相机光轴 world 方向 (= z_camera)
    cycle_added: int               # 哪个 cycle 加入的
    last_gain: float = 0.0         # 上次评估的 volumetric gain (M1.6)
    metadata: dict = field(default_factory=dict)   # 任意扩展


class ExplorationGraph:
    """持久化全局图. 节点不删除, 边在地图变化时可标记为 invalid."""

    def __init__(self):
        self._g = nx.Graph()
        self._tree: Optional[BallTree] = None       # 节点近邻查询
        self._tree_dirty = True                     # 加节点后置 True
        self._node_thetas: list[np.ndarray] = []    # 与 self._g.nodes 顺序一致
        self._node_ids: list[str] = []

    # ---------- 节点 ----------
    def add_node(self, node: GraphNode) -> str:
        """添加节点, 返回 node_id."""
        ...

    def get_node(self, node_id: str) -> GraphNode:
        ...

    def __len__(self) -> int:
        return len(self._g.nodes)

    @property
    def num_edges(self) -> int:
        return self._g.number_of_edges()

    # ---------- 边 ----------
    def add_edge(self, id_a: str, id_b: str, weight: float, metadata: dict = None):
        """已验证的边. weight 是关节空间距离."""
        ...

    def remove_edge(self, id_a: str, id_b: str):
        """地图变化时, 把不再 free 的边删掉."""
        ...

    # ---------- 查询 ----------
    def nodes_within(self, theta: np.ndarray, radius: float) -> list[GraphNode]:
        """关节空间球内的所有节点 (用 BallTree)."""
        ...

    def nearest_node(self, theta: np.ndarray) -> GraphNode:
        """最近邻."""
        ...

    def dijkstra(self, src_id: str, dst_id: str) -> Optional[list[GraphNode]]:
        """最短路径. 返回节点序列, 或 None 若不可达."""
        ...

    def dijkstra_to_many(self, src_id: str, dst_ids: list[str]) -> dict[str, list[GraphNode]]:
        """一对多最短路径. 用单源 Dijkstra."""
        ...

    # ---------- 维护 ----------
    def update_node_gain(self, node_id: str, gain: float, cycle: int):
        """每个 cycle 重新评估节点 gain (M1.8 Global step 用)."""
        ...

    def frontier_nodes(self, gain_threshold: float = 5.0) -> list[GraphNode]:
        """gain 高于阈值的所有节点."""
        ...
```

## 实施步骤

### Step 1: 数据结构和 add_node/add_edge

```python
def add_node(self, node):
    self._g.add_node(node.node_id, data=node)
    self._node_thetas.append(node.theta.copy())
    self._node_ids.append(node.node_id)
    self._tree_dirty = True
    return node.node_id

def add_edge(self, id_a, id_b, weight, metadata=None):
    self._g.add_edge(id_a, id_b, weight=weight, metadata=metadata or {})
```

### Step 2: 近邻查询

BallTree 不支持增量插入，dirty 时 rebuild。

```python
def _maybe_rebuild_tree(self):
    if self._tree_dirty:
        if not self._node_thetas:
            self._tree = None
            return
        arr = np.stack(self._node_thetas)
        self._tree = BallTree(arr, leaf_size=20)
        self._tree_dirty = False

def nodes_within(self, theta, radius):
    self._maybe_rebuild_tree()
    if self._tree is None:
        return []
    idx = self._tree.query_radius(theta.reshape(1, -1), r=radius)[0]
    return [self._g.nodes[self._node_ids[i]]['data'] for i in idx]

def nearest_node(self, theta):
    self._maybe_rebuild_tree()
    if self._tree is None:
        return None
    _, idx = self._tree.query(theta.reshape(1, -1), k=1)
    return self._g.nodes[self._node_ids[idx[0, 0]]]['data']
```

### Step 3: Dijkstra

直接用 networkx：

```python
def dijkstra(self, src_id, dst_id):
    try:
        path_ids = nx.dijkstra_path(self._g, src_id, dst_id, weight='weight')
        return [self._g.nodes[i]['data'] for i in path_ids]
    except nx.NetworkXNoPath:
        return None

def dijkstra_to_many(self, src_id, dst_ids):
    """先单源得到全部最短路径, 再筛."""
    lengths, paths = nx.single_source_dijkstra(self._g, src_id, weight='weight')
    out = {}
    for did in dst_ids:
        if did in paths:
            out[did] = [self._g.nodes[i]['data'] for i in paths[did]]
    return out
```

### Step 4: 边失效

地图更新后，调用方（M1.7）需要重新检查受影响的边。这里只提供基础 API：

```python
def remove_edge(self, id_a, id_b):
    if self._g.has_edge(id_a, id_b):
        self._g.remove_edge(id_a, id_b)

def edges_with_node_in_box(self, theta_center, half_extent):
    """返回所有"两个端点中至少一个在 theta_center 附近"的边. 用于地图局部更新后的边重检."""
    self._maybe_rebuild_tree()
    near_ids = set(n.node_id for n in self.nodes_within(theta_center, half_extent))
    return [(a, b) for a, b in self._g.edges if a in near_ids or b in near_ids]
```

# tests/test_graph.py

def test_add_node():
    g = ExplorationGraph()
    node = GraphNode(
        node_id="n1",
        theta=np.zeros(6),
        ee_pos_world=np.zeros(3),
        ee_dir_world=np.array([0, 0, 1]),
        cycle_added=0,
    )
    g.add_node(node)
    assert len(g) == 1
    assert g.get_node("n1").theta.shape == (6,)


def test_add_edge_and_dijkstra():
    """3 节点的链 a-b-c, dijkstra 返回 [a,b,c]."""
    g = ExplorationGraph()
    a = GraphNode("a", np.zeros(6), np.zeros(3), np.array([0,0,1]), 0)
    b = GraphNode("b", np.ones(6) * 0.1, np.zeros(3), np.array([0,0,1]), 0)
    c = GraphNode("c", np.ones(6) * 0.2, np.zeros(3), np.array([0,0,1]), 0)
    g.add_node(a); g.add_node(b); g.add_node(c)
    g.add_edge("a", "b", weight=1.0)
    g.add_edge("b", "c", weight=1.0)
    
    path = g.dijkstra("a", "c")
    assert [n.node_id for n in path] == ["a", "b", "c"]


def test_dijkstra_finds_shorter_via_third():
    """4 节点: a-b-c, a-d-c. d 路径更短, dijkstra 应选 d."""
    g = ExplorationGraph()
    for i, name in enumerate(["a", "b", "c", "d"]):
        g.add_node(GraphNode(name, np.full(6, i*0.1), np.zeros(3), np.array([0,0,1]), 0))
    g.add_edge("a", "b", weight=10.0)
    g.add_edge("b", "c", weight=10.0)
    g.add_edge("a", "d", weight=1.0)
    g.add_edge("d", "c", weight=1.0)
    path = g.dijkstra("a", "c")
    assert [n.node_id for n in path] == ["a", "d", "c"]


def test_dijkstra_no_path():
    """孤立节点, dijkstra 返回 None."""
    g = ExplorationGraph()
    g.add_node(GraphNode("a", np.zeros(6), np.zeros(3), np.array([0,0,1]), 0))
    g.add_node(GraphNode("b", np.zeros(6), np.zeros(3), np.array([0,0,1]), 0))
    assert g.dijkstra("a", "b") is None


def test_nodes_within():
    """随机撒 100 节点, 球查询."""
    g = ExplorationGraph()
    np.random.seed(0)
    for i in range(100):
        theta = np.random.randn(6) * 0.5
        g.add_node(GraphNode(f"n{i}", theta, np.zeros(3), np.array([0,0,1]), 0))
    
    center = np.zeros(6)
    near = g.nodes_within(center, radius=0.3)
    # 验证: 所有返回的节点都满足距离 ≤ 0.3
    for n in near:
        assert np.linalg.norm(n.theta - center) <= 0.3
    # 验证: 距离 ≤ 0.3 的节点都被返回
    for i in range(100):
        node = g.get_node(f"n{i}")
        d = np.linalg.norm(node.theta - center)
        if d <= 0.3:
            assert node in near


def test_remove_edge_disconnects():
    g = ExplorationGraph()
    a = GraphNode("a", np.zeros(6), np.zeros(3), np.array([0,0,1]), 0)
    b = GraphNode("b", np.ones(6)*0.1, np.zeros(3), np.array([0,0,1]), 0)
    g.add_node(a); g.add_node(b)
    g.add_edge("a", "b", weight=1.0)
    assert g.dijkstra("a", "b") is not None
    g.remove_edge("a", "b")
    assert g.dijkstra("a", "b") is None


def test_frontier_nodes():
    g = ExplorationGraph()
    for i in range(5):
        n = GraphNode(f"n{i}", np.full(6, i*0.1), np.zeros(3), np.array([0,0,1]), 0)

def test_persistent_across_cycles():

### 性能测试

```python
def test_dijkstra_scales():
    """1000 节点的图, dijkstra < 50 ms."""
    g = ExplorationGraph()
    np.random.seed(0)
    nodes = [GraphNode(f"n{i}", np.random.randn(6) * 0.5,
                       np.zeros(3), np.array([0,0,1]), 0)
             for i in range(1000)]
    for n in nodes:
        g.add_node(n)
    # 每个节点连 5 个最近邻
    for n in nodes:
        for nb in g.nodes_within(n.theta, 0.5)[:5]:
            if nb.node_id != n.node_id:
                w = float(np.linalg.norm(nb.theta - n.theta))
                g.add_edge(n.node_id, nb.node_id, weight=w)
    
    import time
    t = time.time()
    for _ in range(20):
        path = g.dijkstra("n0", f"n{999}")
    elapsed = (time.time() - t) / 20 * 1000
    print(f"Dijkstra 1000-node: {elapsed:.2f} ms")
    assert elapsed < 50
```
